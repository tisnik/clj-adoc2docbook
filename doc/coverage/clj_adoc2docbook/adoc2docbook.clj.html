020  (defn escape-xml-chars
021    "Escape characters with special meaning in XML."
022    [string]
023    (str/escape string
024      {\< "&lt;", \> "&gt;", \& "&amp;"}))
025  
026  ; Some special characters have special meaning:
027  ; <systemitem> - rendered as bold, suggest using * (asterisk)
028  ; <package>    - rendered as italic, suggest using _ (underscore)
029  ; <literal>    - rendered as bold, suggest using ` (backtick)
030  ; <command>    - rendered as bold, sugget using " (double quote)
031  (defn asciidoc-like-transformation
032    [link-to-bugzilla string]
033    ; TODO - just a quick hack, would be better to replace with proper state
034    ; machine
035    (-> string
036        ; special handling for words at the beginning of line
037        ;(str/replace #"^'([^\s]+\s*[^\s']*)'([ .,])"    #(str "<literal>"    (second %1) "</literal>" (nth %1 2)))
038        (str/replace #"^`([^\s]+\s*[^\s`]*)`([ .,(])"    #(str "<literal>"    (second %1) "</literal>" (nth %1 2)))
039        (str/replace #"^\"([^\s]+\s*[^\s\"]*)\"([ .,])" #(str "<command>"    (second %1) "</command>" (nth %1 2)))
040        (str/replace #"^\*([^\s]+\s*[^\s\*]*)\*([ .,])" #(str "<systemitem>" (second %1) "</systemitem>" (nth %1 2)))
041        (str/replace #"^_([^\s]+\s*[^\s\_]*)_([ .,])"   #(str "<package>"    (second %1) "</package>" (nth %1 2)))
042  
043        ; words in the middle - should start with <space> and end with <space>, <dot>, <comma>, or <parathesis>
044        ;(str/replace  #" '([^\s]+(\s+[^\s']+)*)'([ .,)])"    #(str " <literal>"    (second %1) "</literal>" (nth %1 3)))
045        (str/replace  #"([ (])`([^\s]+(\s+[^\s`]+)*)`([ .,()])"    #(str (second %1) "<literal>" (nth %1 2) "</literal>" (nth %1 4)))
046        (str/replace #" \"([^\s]+(\s+[^\s\"]+)*)\"([ .,)])"   #(str " <command>"     (second %1) "</command>" (nth %1 3)))
047        (str/replace #" \*([^\s]+(\s+[^\s\*]+)*)\*([ .,\-)])"   #(str " <systemitem>"  (second %1) "</systemitem>" (nth %1 3)))
048        (str/replace  #" _([^\s]+(\s+[^\s_]+)*)_([ .,:)])"    #(str " <package>"    (second %1) "</package>" (nth %1 3)))
049  
050        ; special handling for words at end of line
051        ;(str/replace  #" '([^\s]+\s*[^\s']*\s*[^\s']*)'$"    #(str " <literal>"    (second %1) "</literal>"))
052        (str/replace  #" `([^\s]+\s*[^\s`]*\s*[^\s`]*)`$"    #(str " <literal>"    (second %1) "</literal>"))
053        (str/replace #" \"([^\s]+\s*[^\s\"]*\s*[^\s\"]*)\"$" #(str " <command>"    (second %1) "</command>"))
054        (str/replace #" \*([^\s]+\s*[^\s\*]*\s*[^\s\*]*)\*$" #(str " <systemitem>" (second %1) "</systemitem>"))
055        (str/replace  #" _([^\s]+\s*[^\s_]*\s*[^\s_]*)_$"    #(str " <package>"    (second %1) "</package>"))
056  
057        ; special cases
058        (str/replace  #"\(_([^\s]+(\s+[^\s_]+)*)_\)"    #(str "(<package>"    (second %1) "</package>)"))
059  
060        ; for situations when whole line is in backticks
061        (str/replace  #"^`([^`]+)`$"    #(str "<literal>" (second %1) "</literal>"))
062  
063        ; BZ#123456 should be rendered as <ulink>
064        (str/replace #"([^,]) (BZ#[0-9]+)" #(str (second %1) " <ulink url='" link-to-bugzilla (subs (third %1) 3) "'>" (third %1) "</ulink>"))))

066  (defn get-line-type
067    "There exist four line types, according to the specification:
068       1: normal lines
069       2: lines for the <screen> environment
070       3: lines for the <itemizedlist> environment
071       4: empty lines that should be ignored when not in <screen> environment"
072    [line]
073    (cond (empty? line)                   :empty
074          (empty? (str/trim line))        :empty
075          (re-matches #"\s*\*\s+.*" line) :item   ; !!!order is important!!!
076          (.startsWith line "   ")        :screen ; 3 or more spaces
077          :else                           :normal))

081    (let [star-index (inc (.indexOf line "*"))

085           "</para></listitem>\n")))

181  (defn get-output
182    [link-to-bugzilla line last-status]
183    (condp = [last-status (get-line-type line)]
184      [:nothing         :empty]    [:nothing         nil] ; ignore
185      [:nothing         :normal]   [:paragraph       (start-para link-to-bugzilla line)]
186      [:nothing         :item]     [:itemized-list   (first-list-item link-to-bugzilla line)]
187      [:nothing         :screen]   [:screen          (screen-begin line)]
188      [:paragraph       :normal]   [:paragraph       (middle-in-para link-to-bugzilla line)]
189      [:paragraph       :empty]    [:nothing         (end-para)]
190      [:paragraph       :item]     [:itemized-list   (end-para-first-list-item link-to-bugzilla line)]
191      [:paragraph       :screen]   [:screen          (end-para-screen-begin line)]
192      [:itemized-list   :item]     [:itemized-list   (list-item link-to-bugzilla line)]
193      [:itemized-list   :empty]    [:itemized-list   nil] ;ignore
194      [:itemized-list   :normal]   [:paragraph       (last-list-item link-to-bugzilla line)]
195      [:itemized-list   :screen]   [:itemized-screen (screen-list-item-begin line)]
196      [:itemized-screen :screen]   [:itemized-screen (screen-in-middle line)]
197      [:itemized-screen :item]     [:itemized-list   (screen-list-item-end link-to-bugzilla line)]
198      [:itemized-screen :empty]    [:itemized-screen nil]
199      [:itemized-screen :normal]   [:paragraph       (screen-end link-to-bugzilla line)]
200      [:screen          :screen]   [:screen          (screen-in-middle-notrim line)]
201      [:screen          :normal]   [:paragraph       (screen-end link-to-bugzilla line)]
202      [:screen          :empty]    [:screen-empty    nil]
203      [:screen-empty    :screen]   [:screen          (screen-empty line)]
204      [:screen-empty    :normal]   [:paragraph       (screen-end link-to-bugzilla line)]
205      [:screen-empty    :empty]    [:screen-empty    nil]
206                                   [last-status line]))

213      :paragraph       "</para>\n"

218  (def http-regexp
219    ; scheme     host name                             domain       query+fragment               we don't want those chars at the end of URL
220    #"(https?:\/\/(www\.)?[-a-zA-Z0-9@:%._\+~#=]{2,256}\.[a-z]{2,6}\b([-a-zA-Z0-9@:%_\+.~#?&//=]*)[^?.,!) ])")

222  (def bz-href-regexp
223    #"BZ####([0-9]+)")

225  (defn create-normal-links
226    [string]
227    (let [href (re-find http-regexp string)]
228      (if href
229        (str/replace string
230                     http-regexp
231                     #(str "<ulink url='" (first %1) "'>" (first %1) "</ulink>"))
232        string)))

234  (defn create-bugzilla-links
235    [link-to-bugzilla string]
236    (let [bz-href (re-find bz-href-regexp string)]
237      (if bz-href
238        (str/replace string
239                     bz-href-regexp
240                     #(str "BZ#<ulink url='" link-to-bugzilla (second %1) "'>" (second %1) "</ulink>"))
241        string)))

243  (defn create-links
244    [link-to-bugzilla string]
245    (->> string
246         create-normal-links
247         (create-bugzilla-links link-to-bugzilla)))

261  (defn transform-lines
262    [link-to-bugzilla lines]
263    (loop [result []
264           status :nothing
265           lines lines]
266      (let [line                (first lines)
267            [new-status output] (get-output link-to-bugzilla line status)
268            rest                (next lines)
269            new-result
270              (if output
271                (if (vector? output)
272                    (into result output)
273                    (conj result output))
274                result)]
275        ;(println status  (get-line-type line) new-status line)
276        (if rest
277          (recur new-result
278                 new-status rest)
279                 (conj new-result (closing-tag new-status))))))
